const express = require('express');
const mongoose = require('mongoose');

const app = express();
const PORT = 3001;

mongoose.connect(MONGO_URI)
  .then(() => console.log('âœ… Connected to MongoDB'))
  .catch((err) => console.error('âŒ MongoDB connection error:', err));

// Enable JSON parsing middleware
app.use(express.json());

const rawDataSchema = new mongoose.Schema({}, { strict: false });
const RawData = mongoose.model('RawData', rawDataSchema, 'raw_data');

app.get('/raw-data', async (req, res) => {
  try {
    const raw_data = await RawData.find();
    res.json(raw_data);
  } catch (err) {
    console.error('Error fetching data:', err);
    res.status(500).json({ error: 'Failed to fetch data' });
  }
});

// Health check endpoint
app.get('/health', (req, res) => {
  res.json({ status: 'OK', timestamp: new Date().toISOString() });
});

// Handle graceful shutdown
process.on('SIGINT', async () => {
  console.log('ğŸ”„ Shutting down gracefully...');
  await mongoose.connection.close();
  console.log('âœ… MongoDB connection closed');
  process.exit(0);
});

app.listen(PORT, () => {
  console.log(`ğŸš€ Server is running on http://localhost:${PORT}`);
  console.log(`ğŸ“Š Raw data endpoint: http://localhost:${PORT}/raw-data`);
  console.log(`â¤ï¸  Health check: http://localhost:${PORT}/health`);
});

// Export for testing
module.exports = app;
